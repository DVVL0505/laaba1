#include <stdio.h>
#include <stdlib.h>
#include <string.h>
#include <conio.h>

int main()
{
    system("chcp 65001");
    system("cls");
    static int mas[256];
    char *str = "qweqqq asd zxc";
    int i, max=0, symbol;
    for (i = 0; i < strlen(str); i++)
    {
       if (str[i] != ' ' && str[i] != '\n' && str[i] != '\t')
          mas[str[i]]++;
    }
    max=mas[0];
    for (i = 1; i < 256; i++)
    {
       if (max < mas[i])
       {
          max = mas[i];
          symbol = i;
       }
    }
    printf("Буква %c встречается наибольшее количество раз - %d", symbol, max);
}
